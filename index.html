#include // Definire pini const int servoPin = 9; // Pinul pentru servomotor const int sensorPin = A0; // Pinul analogic pentru senzorul de umiditate/apă // Setări praguri const int pragDetectie = 500; // Ajustează valoarea în funcție de sensibilitatea senzorului Servo bratServo; // Crearea obiectului Servo void setup() { // Inițializare comunicare serială pentru monitorizare Serial.begin(9600); // Atașarea servomotorului la pinul configurat bratServo.attach(servoPin); // Setare poziție inițială (0 grade) bratServo.write(0); Serial.println("Sistem pornit. Aștept detectare..."); } void loop() { // Citire valoare de la senzor int valoareSenzor = analogRead(sensorPin); // Afișare valoare în Serial Monitor (ajută la calibrare) Serial.print("Valoare senzor: "); Serial.println(valoareSenzor); // Verificare condiție if (valoareSenzor > pragDetectie) { Serial.println("Obiect/Apă detectată! Mișcare servo la 90 grade."); // Mută servomotorul la 90 de grade bratServo.write(90); // Menține poziția (de exemplu, 2 secunde) delay(2000); } else { // Dacă nu detectează nimic, revine la 0 grade bratServo.write(0); } // Mică pauză pentru stabilitatea citirilor delay(100); }
